class rainDrops:
    def pour(self, n:int)->str:
        out  = ""
        if n%3==0:
            out += "Pling"
        if n%5==0:
            out += "Plang"
        if n%7==0:
            out += "Plong"
        if out == "":
            out = f"{n}"
        return out

test = rainDrops()
print(test.pour(28))
test = rainDrops()
print(test.pour(30))
test = rainDrops()
print(test.pour(34))
