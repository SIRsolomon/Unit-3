```.py
from kivymd.app import MDApp

class quiz38(MDApp):
    def build(self):
        self.In =None
        return

    def mysteryPageA(self):
        self.In = 'This is message A'
        self.root.ids.click_label.text = f"{self.In}"

    def mysteryPageB(self):
        self.In = 'This is message B'
        self.root.ids.click_label.text = f"{self.In}"

t = quiz38()
t.run()
```
