```.py
class palNum:
    def __init__(self, A, B):
        self.A = A
        self.B = B

    def get_pal_list(self):
        out = []
        for num in range(self.A, self.B + 1):
            if str(num) == str(num)[::-1]:
                out.append(num)
        return out

test = palNum(1,9)
print(test.get_pal_list())

test = palNum(10,199)
print(test.get_pal_list())
```
